Задача

Реализовать параметризованный конструктор класса Point так, чтобы при выполнении данных сценариев на экран был выведен указанный результат.
Номер по порядку Сценарий Результат
Point p = new Point(53, 59); Console.WriteLine(p.Print()); (53;59)
Point p1 = new Point(53, 59); Point p2 = new Point(61,67); Console.WriteLine("Distance between " + p1.Print() + " and " + p2.Print() + " is " + p1.DistanceTo(p2)); Distance between (53;59) and (61;67) is 11.3137
double dist = new Point(79, 83) .DistanceTo(new Point(89, 97)); Console.WriteLine(dist); 8.4853
Point p = new Point(71, 73); Console.WriteLine("Distance between " + p.Print() + " to zero is " + p.DistanceTo(new Point(0, 0))); Distance between (71;73) to zero is 101.8332
Console.WriteLine(new Point(79, 83) .DistanceTo(new Point(89, 97))); 17.2047
double dist = Point.DistanceBetween( new Point(151, 157), new Point(163, 167)); Console.WriteLine(dist); 15.6204
Console.WriteLine(Point.DistanceBetween( new Point(173, 179), new Point(181, 191))); 14.4222
