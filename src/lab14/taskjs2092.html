Задача

Инициализировать объект interval так, чтобы при выполнении данных сценарев на экран был выведен указанный результат.
Номер по порядку Сценарий Результат
interval.a = 0; interval.b = 10; bool fact = interval.contains(5); if (fact) { console.log("Содержит"); } else { console.log("Не содержит"); } Содержит
interval.a = 100; interval.b = 1000; if (interval.contains(100)) { console.log("Содержит"); } else { console.log("Не содержит"); } Содержит
interval.a = 13; interval.b = 17; if (interval.contains(17)) { console.log("Содержит"); } else { console.log("Не содержит"); } Содержит
interval.a = 23; interval.b = 29; if (interval.contains(0)) { console.log("Содержит"); } else { console.log("Не содержит"); } Не содержит
interval.a = 31; interval.b = 43; if (interval.contains(50)) { console.log("Содержит"); } else { console.log("Не содержит"); } Не содержит
interval.a = 100; interval.b = 0; try { interval.contains(10); } catch (e) { console.log(e.Message); } Корректность интервала была нарушена
