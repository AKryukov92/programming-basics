Задача
Реализовать класс Rectangle так, чтобы при выполнении данных сценариев на экран был выведен указанный результат с точностью 0.0001.
Номер по порядку Сценарий Результат
Rectangle r = new Rectangle(5,5,20,20); Point p = new Point(30,30); Console.WriteLine(r.Contains(p)); false
Point corner = new Point(10,10); Rectangle r = new Rectangle(corner,20,20); Point p = new Point(30,30); Console.WriteLine(r.Contains(p)); true
Point corner = new Point(10,10); Rectangle r = new Rectangle(corner,20,20); Console.WriteLine(r.Area); 400
Interval projectionX = new Interval(15, 30); Interval projectionY = new Interval(20, 35); Rectangle r = new Rectangle(projectionX, projectionY); Console.WriteLine(r.Area); 225
Rectangle r = new Rectangle(100, 200, 300, 400); Interval i = new Interval(50, 75); Console.WriteLine(r.ProjectionX.Intersects(i)); false
Rectangle r = new Rectangle(100, 200, 300, 400); Interval i = new Interval(100, 300); Console.WriteLine(r.ProjectionY.Intersects(i)); true
