Задача

Реализовать класс Point так, чтобы при выполнении данного сценария на экран был выведен указанный результат c точностью 0.0001.
Номер по порядку Сценарий Результат
Point p = new Point(); p.x = 0; p.y = 0; double dist = p.DistanceTo(4,3); Console.WriteLine(dist); 5.0
Point p = new Point(); p.x = 0; p.y = 0; Console.WriteLine(p.DistanceTo(0,0)); 0.0
Point p = new Point(); p.x = 19; p.y = 23; Console.WriteLine(p.DistanceTo(11,13)); 12.8062
Point p = new Point(); p.x = 11; p.y = 23; Console.WriteLine(p.DistanceTo(11,13)); 10.0
