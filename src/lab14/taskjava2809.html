Задача

Дополнить класс Point так, чтобы при выполнении данных сценариев на экран был выведен указанный результат с точностью 0.0001
Номер по порядку Сценарий Результат
Point p = new Point(); p.x = 127; p.y = 113; String str = p.print(); System.out.println(str); (127;113)
Point p = new Point(); p.x = 31; p.y = 37; System.out.println(p.print()); (31;37)
Point p = new Point(); p.setCoordinates(5,7); System.out.println(p.print()); (5;7)
Point p1 = new Point(); p1.setCoordinates(23,29); System.out.println("Distance between " + p1.print() + " to (0;0) is " + p1.distanceTo(0,0)); Distance between (23;29) to (0;0) is 37.0135
Point p1 = new Point(); p1.setCoordinates(11,13); Point p2 = new Point(); p2.x = 17; p2.y = 19; System.out.println("Distance between " + p1.print() + " and " + p2.print() + " is " + p1.distanceTo(p2)); Distance between (11;13) and (17;19) is 8.4852
