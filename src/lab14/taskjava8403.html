Задача
Реализовать класс Point так, чтобы при выполнении данного сценария на экран был выведен указанный результат c точностью 0.0001.
Номер по порядку Сценарий Результат
Point p = new Point(); p.x = 0; p.y = 0; double dist = p.distanceTo(4,3); System.out.println(dist); 5.0
Point p = new Point(); p.x = 0; p.y = 0; System.out.println(p.distanceTo(0,0)); 0.0
Point p = new Point(); p.x = 19; p.y = 23; System.out.println(p.distanceTo(11,13)); 12.8062
Point p = new Point(); p.x = 11; p.y = 23; System.out.println(p.distanceTo(11,13)); 10.0
