Задача

Дополнить класс Point так, чтобы при выполнении данных сценарев на экран был выведен указанный результат c точностью 0.0001.
Номер по порядку Сценарий Результат
Point p1 = new Point() { x = 109, y = 113 }; Point p2 = new Point(); p2.x = 127; p2.y = 131; double dist = Point.DistanceBetween(p1, p2); Console.WriteLine("{0:F4}", dist); 25.4558
Point p1 = new Point(); p1.x = 137; p1.y = 139; Point p2 = new Point() { x = 139, y = 149 }; Console.WriteLine("{0:F4}", Point.DistanceBetween(p1, p2)); 10.1980
Point p1 = new Point(); p1.x = 151; p1.y = 157; Point p2 = null; try { Console.WriteLine("{0:F4}", Point.DistanceBetween(p1, p2)); } catch (ArgumentException e) { Console.WriteLine(e.Message); } Начальная точка не инициализирована
Point p1 = null; Point p2 = new Point() { x = 163, y = 167 }; try { Console.WriteLine("{0:F4}", Point.DistanceBetween(p1,p2)); } catch (ArgumentException e) { Console.WriteLine(e.Message); } Конечная точка не инициализирована
