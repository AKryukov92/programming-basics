Задача

Дополнить класс Point так, чтобы при выполнении данных сценариев на экран был выведен указанный результат с точностью 0.0001
Номер по порядку Сценарий Результат
Point p = new Point(); p.x = 127; p.y = 113; String str = p.Print(); Console.WriteLine(str); (127;113)
Point p = new Point(); p.x = 31; p.y = 37; Console.WriteLine(p.Print()); (31;37)
Point p = new Point(); p.SetCoordinates(5,7); Console.WriteLine(p.Print()); (5;7)
Point p1 = new Point(); p1.SetCoordinates(23,29); Console.WriteLine("Distance between " + p1.Print() + " to (0;0) is " + p1.DistanceTo(0,0)); Distance between (23;29) to (0;0) is 37.0135
Point p1 = new Point(); p1.SetCoordinates(11,13); Point p2 = new Point(); p2.x = 17; p2.y = 19; Console.WriteLine("Distance between " + p1.Print() + " and " + p2.Print() + " is " + p1.DistanceTo(p2)); Distance between (11;13) and (17;19) is 8.4852
