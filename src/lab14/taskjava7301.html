Задача

Реализовать параметризованный конструктор класса Point так, чтобы при выполнении данных сценариев на экран был выведен указанный результат.
Номер по порядку Сценарий Результат
Point p = new Point(53, 59); System.out.println(p.print()); (53;59)
Point p1 = new Point(53, 59); Point p2 = new Point(61,67); System.out.println("Distance between " + p1.print() + " and " + p2.print() + " is " + p1.distanceTo(p2)); Distance between (53;59) and (61;67) is 11.3137
double dist = new Point(79, 83) .distanceTo(new Point(89, 97)); System.out.println(dist); 8.4853
Point p = new Point(71, 73); System.out.println("Distance between " + p.print() + " to zero is " + p.distanceTo(new Point(0, 0))); Distance between (71;73) to zero is 101.8332
System.out.println(new Point(79, 83) .distanceTo(new Point(89, 97))); 17.2047
double dist = Point.distanceBetween( new Point(151, 157), new Point(163, 167)); System.out.println(dist); 15.6204
System.out.println(Point.distanceBetween( new Point(173, 179), new Point(181, 191))); 14.4222
